\SingleHeader{010-OntologyContext\_ClassesHierarchy}
\CONTEXT{010-OntologyContext\_ClassesHierarchy}{}{}
\SETS{
	\Set{C}{l'ensemble des classes de l'ontologie}
}
\CONSTANTS{
	\Constant{thing}{la racine de l'arbre des classes}
	\Constant{C\_leaves}{les feuilles de la hiérarchie des classes}
	\Constant{has\_parent}{la relation directe entre une classe enfant et une ou des classes parentes, servant à la hiérarchie des classes}
}
\AXIOMS{
	\Axiom{onto\_finite\_C}{false}{$finite(C)$}{\\C est un ensemble fini}
	\Axiom{onto\_type\_thing}{false}{$thing \in{} C$}{\\thing est une classe}
	\Axiom{onto\_type\_CLeaves}{false}{$C\_leaves \subseteq{} C$}{\\certaines classes sont des feuilles}
	\Axiom{onto\_type\_hasParent}{false}{$has\_parent \in{} C \setminus{} \{thing\} \strel{} C \setminus{} C\_leaves$}{\\has$\_$parent est une relation des classes sans la racine Thing dans l'ensemble des classes privé des feuilles. Toute classe, sauf thing, a au moins un parent et toutes classes, sauf les feuilles, sont parents d'au moins une classe.}
	\Axiom{onto\_C\_axm\_treeInduction}{false}{$\\\forall{}S \qdot{} S\subseteq{}C \land{} \\~    thing\in{}S \land{} \\~   (\forall{}n \qdot{} n\in{}C\setminus{}\{thing\} \land{} has\_parent[\{n\}] \subseteq{} S \limp{} n\in{}S) \\~  \limp{} \\~    C \subseteq{} S$}{\\la relation has$\_$parent a une structure d'arbre à partir de thing}
	\Axiom{onto\_C\_axm\_antiReflexive}{false}{$\forall{}c \qdot{} c \in{} C \limp{} c \notin{} has\_parent[\{c\}]$}{\\la relation has$\_$parent est anti-réflexive: un noeud n'est pas parent de lui-même}
	\Axiom{onto\_C\_thm\_thingInLeaves}{true}{$thing \in{} C\_leaves \limp{} C = \{thing\} \land{} has\_parent=\emptyset{}$}{\\si thing est une feuille, il s'agit de la seule classe de l'ontologie et la relation parent est vide}
}
\END
